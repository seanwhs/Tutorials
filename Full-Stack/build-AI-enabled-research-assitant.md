# Building Your Own AI-Powered Research Assistant

**A Step-by-Step Tutorial**

Hello! Welcome to the ultimate guide for building a **production-grade AI Research Assistant**.

We’ll transform a basic RAG system into a **truly intelligent, agentic research tool** by adding:
- **Multi-file support** — upload entire folders of PDFs at once
- **Better chunking with LangChain** — smarter, context-preserving splits
- **Agentic capabilities** — the assistant can reason, use tools (web search, calculator, etc.)
- **Conversation memory** — multi-turn, coherent chats
- **Document summarization** — generate quick overviews of long papers
- **Authentication** with Supabase Auth
- **Document list UI** with summary buttons
- **Helicone observability** for monitoring

Everything stays **secure** with PostgreSQL Row Level Security (RLS), production-ready, and explained as if you’re seeing it for the first time.

## Step 1: Supabase Schema

Run these SQL commands in your Supabase Dashboard → SQL Editor:

```sql
-- Documents table
create table documents (
  id bigint generated by default as identity primary key,
  name text not null,
  owner_id uuid references auth.users not null,
  status text default 'parsing',
  summary text,
  created_at timestamp default now()
);

-- Document sections (chunks + embeddings)
create table document_sections (
  id bigint generated by default as identity primary key,
  document_id bigint references documents on delete cascade,
  content text,
  embedding vector(1536),
  chunk_index integer,
  owner_id uuid default auth.uid()
);

-- Conversation memory
create table conversation_messages (
  id bigint generated by default as identity primary key,
  conversation_id text not null,
  owner_id uuid references auth.users not null,
  role text not null,
  content text not null,
  created_at timestamp default now()
);

-- Enable RLS
alter table documents enable row level security;
alter table document_sections enable row level security;
alter table conversation_messages enable row level security;

-- Basic RLS policies
create policy "Users can view own documents" on documents for select using (owner_id = auth.uid());
create policy "Users can insert own documents" on documents for insert with check (owner_id = auth.uid());
create policy "Users can view own sections" on document_sections for select using (owner_id = auth.uid());
create policy "Users can manage own conversations" on conversation_messages using (owner_id = auth.uid());
```

## Step 2: Enhanced Ingestion Pipeline (Multi-File + LangChain Chunking)

Create `app/api/ingest/route.ts`:

```typescript
import { createClient } from '@supabase/supabase-js';
import { NextRequest, NextResponse } from 'next/server';
import OpenAI from 'openai';
import { RecursiveCharacterTextSplitter } from "langchain/text_splitter";

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export async function POST(req: NextRequest) {
  try {
    const { files, userId } = await req.json();

    const results = [];

    for (const file of files) {
      const { fileBuffer, fileName } = file;

      const { data: doc } = await supabase
        .from('documents')
        .insert({ name: fileName, owner_id: userId, status: 'parsing' })
        .select()
        .single();

      const buffer = Buffer.from(fileBuffer, 'base64');
      const pdfParse = (await import('pdf-parse')).default;
      const { text } = await pdfParse(buffer);

      console.log(`📄 Extracted ${text.length} characters from ${fileName}`);

      const splitter = new RecursiveCharacterTextSplitter({
        chunkSize: 1000,
        chunkOverlap: 200,
        separators: ["\n\n", "\n", ". ", " ", ""],
      });

      const chunks = await splitter.splitText(text);

      await supabase.from('documents').update({ status: 'embedding' }).eq('id', doc.id);

      const embeddings = await generateEmbeddings(chunks);

      const sections = chunks.map((content: string, i: number) => ({
        document_id: doc.id,
        content,
        embedding: embeddings[i],
        chunk_index: i,
        owner_id: userId
      }));

      await supabase.from('document_sections').insert(sections);

      await supabase.from('documents').update({ status: 'completed' }).eq('id', doc.id);

      results.push({ docId: doc.id, fileName });
    }

    return NextResponse.json({ success: true, processed: results });
  } catch (error: any) {
    console.error(error);
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

async function generateEmbeddings(texts: string[]) {
  const response = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: texts,
  });
  return response.data.map(item => item.embedding);
}
```

**Why LangChain Chunking Wins**: It intelligently splits on natural boundaries while preserving context.

## Step 3: Multi-File Upload UI with Progress

Create `components/FileUploader.tsx`:

```tsx
'use client';

import { useState } from 'react';
import { useQuery } from '@tanstack/react-query';
import { createClient } from '@supabase/supabase-js';
import { Progress } from '@/components/ui/progress';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
);

export default function FileUploader() {
  const [docIds, setDocIds] = useState<string[]>([]);
  const [uploading, setUploading] = useState(false);

  const handleMultiUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const selectedFiles = Array.from(e.target.files || []);
    if (!selectedFiles.length) return;

    setUploading(true);
    const { data: { user } } = await supabase.auth.getUser();
    const userId = user?.id;

    const filePromises = selectedFiles.map(file => 
      new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = () => {
          const base64 = (reader.result as string).split(',')[1];
          resolve({ fileBuffer: base64, fileName: file.name });
        };
        reader.readAsDataURL(file);
      })
    );

    const fileData = await Promise.all(filePromises);

    const res = await fetch('/api/ingest', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ files: fileData, userId })
    });

    const result = await res.json();
    setDocIds(result.processed.map((p: any) => p.docId));
    setUploading(false);
  };

  const { data: docs } = useQuery({
    queryKey: ['documents', docIds],
    queryFn: async () => {
      if (!docIds.length) return [];
      const { data } = await supabase
        .from('documents')
        .select('id, name, status')
        .in('id', docIds);
      return data || [];
    },
    enabled: docIds.length > 0,
    refetchInterval: 2000,
  });

  return (
    <div className="max-w-2xl mx-auto p-8 border rounded-2xl bg-white shadow">
      <h2 className="text-3xl font-bold mb-6">📚 Upload Research Documents</h2>
      <input
        type="file"
        multiple
        accept=".pdf"
        onChange={handleMultiUpload}
        disabled={uploading}
        className="block w-full text-sm text-gray-500 file:mr-4 file:py-3 file:px-6 file:rounded-xl file:bg-blue-50"
      />

      {docs?.length > 0 && (
        <div className="mt-8 space-y-4">
          <h3 className="font-semibold">Processing {docs.length} document(s):</h3>
          {docs.map((doc: any) => (
            <div key={doc.id} className="border p-4 rounded-lg">
              <p className="font-medium">{doc.name}</p>
              <Progress 
                value={doc.status === 'completed' ? 100 : doc.status === 'embedding' ? 70 : 35} 
                className="mt-2"
              />
              <p className="text-xs text-gray-500 mt-1 capitalize">{doc.status}</p>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

## Step 4: Agentic Research Assistant (Tools + Memory)

Create `app/api/agent/route.ts`:

```typescript
import { createClient } from '@supabase/supabase-js';
import { NextRequest, NextResponse } from 'next/server';
import { initializeAgentExecutorWithOptions } from "langchain/agents";
import { ChatOpenAI } from "@langchain/openai";
import { Calculator } from "langchain/tools/calculator";
import { TavilySearchResults } from "@langchain/community/tools/tavily_search";
import OpenAI from 'openai';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

const llm = new ChatOpenAI({
  modelName: "gpt-4o-mini",
  temperature: 0.2,
});

export async function POST(req: NextRequest) {
  try {
    const { message, userId, conversationId = "default" } = await req.json();

    const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
    const embeddingResponse = await openai.embeddings.create({
      model: "text-embedding-3-small",
      input: message,
    });
    const queryEmbedding = embeddingResponse.data[0].embedding;

    const { data: chunks } = await supabase.rpc('hybrid_search', {
      query_text: message,
      query_embedding: queryEmbedding,
      match_threshold: 0.75,
      match_count: 7
    });

    const context = chunks?.map((c: any) => c.content).join("\n\n---\n\n") || "";

    const tools = [
      new Calculator(),
      new TavilySearchResults({ maxResults: 3, apiKey: process.env.TAVILY_API_KEY }),
    ];

    const executor = await initializeAgentExecutorWithOptions(tools, llm, {
      agentType: "chat-conversational-react-description",
      verbose: true,
    });

    const input = `
You are an expert research assistant with access to the user's private documents.

=== RELEVANT DOCUMENT CONTEXT ===
${context}
=== END CONTEXT ===

Use the documents when possible. Use tools for math, current events, or missing info.
Always be accurate and helpful.

User Question: ${message}
`;

    const result = await executor.call({
      input,
      chat_history: await getConversationHistory(conversationId, userId)
    });

    await saveConversationMessage(conversationId, userId, message, result.output);

    return NextResponse.json({
      answer: result.output,
      sources: chunks
    });
  } catch (error: any) {
    console.error(error);
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}
```

## Step 5: Conversation Memory Helpers (`lib/conversation.ts`)

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export async function getConversationHistory(conversationId: string, userId: string) {
  const { data } = await supabase
    .from('conversation_messages')
    .select('role, content')
    .eq('conversation_id', conversationId)
    .eq('owner_id', userId)
    .order('created_at', { ascending: true })
    .limit(12);

  return data?.map(msg => `${msg.role}: ${msg.content}`).join("\n") || "";
}

export async function saveConversationMessage(
  conversationId: string,
  userId: string,
  userMessage: string,
  assistantResponse: string
) {
  await supabase.from('conversation_messages').insert([
    { conversation_id: conversationId, owner_id: userId, role: 'user', content: userMessage },
    { conversation_id: conversationId, owner_id: userId, role: 'assistant', content: assistantResponse }
  ]);
}
```

## Step 6: Document Summarization

Create `app/api/summarize/route.ts`:

```typescript
import { createClient } from '@supabase/supabase-js';
import { NextRequest, NextResponse } from 'next/server';
import { loadSummarizationChain } from "langchain/chains";
import { RecursiveCharacterTextSplitter } from "langchain/text_splitter";
import { ChatOpenAI } from "@langchain/openai";

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

const llm = new ChatOpenAI({ modelName: "gpt-4o-mini", temperature: 0.3 });

export async function POST(req: NextRequest) {
  try {
    const { documentId, userId } = await req.json();

    const { data: doc } = await supabase
      .from('documents')
      .select('*')
      .eq('id', documentId)
      .eq('owner_id', userId)
      .single();

    if (!doc) return NextResponse.json({ error: "Not found" }, { status: 404 });

    const { data: sections } = await supabase
      .from('document_sections')
      .select('content')
      .eq('document_id', documentId)
      .order('chunk_index');

    const fullText = sections?.map(s => s.content).join("\n\n") || "";

    const splitter = new RecursiveCharacterTextSplitter({ chunkSize: 4000, chunkOverlap: 400 });
    const docs = await splitter.createDocuments([fullText]);

    const chain = loadSummarizationChain(llm, { type: "map_reduce", verbose: true });
    const { text: summary } = await chain.call({ input_documents: docs });

    await supabase.from('documents').update({ summary }).eq('id', documentId);

    return NextResponse.json({ summary, documentName: doc.name });
  } catch (error: any) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}
```

## Step 7: Beautiful Agent Chat UI with Memory

Create `components/ResearchAgentChat.tsx`:

```tsx
'use client';
import { useState } from 'react';

export default function ResearchAgentChat() {
  const [messages, setMessages] = useState<Array<{role: string; content: string}>>([]);
  const [input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);

  const sendMessage = async () => {
    if (!input.trim() || isLoading) return;

    const userMsg = { role: 'user', content: input };
    setMessages(prev => [...prev, userMsg]);
    setInput("");
    setIsLoading(true);

    const res = await fetch('/api/agent', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        message: input,
        userId: "your-user-id", // replace with real auth
        conversationId: "research-chat-1"
      })
    });

    const data = await res.json();
    setMessages(prev => [...prev, { role: 'assistant', content: data.answer }]);
    setIsLoading(false);
  };

  return (
    <div className="flex flex-col h-[700px] border rounded-3xl p-6 bg-gray-50">
      <div className="flex-1 overflow-y-auto space-y-6 pr-4">
        {messages.map((msg, i) => (
          <div key={i} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
            <div className={`max-w-[85%] p-5 rounded-2xl ${msg.role === 'user' ? 'bg-blue-600 text-white' : 'bg-white shadow-sm'}`}>
              {msg.content}
            </div>
          </div>
        ))}
        {isLoading && <p className="text-blue-600">Thinking... (using tools if needed)</p>}
      </div>

      <div className="flex gap-3 mt-6">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && sendMessage()}
          className="flex-1 p-4 border rounded-2xl focus:outline-none focus:ring-2 focus:ring-blue-500"
          placeholder="Ask anything — math, web search, or about your documents..."
        />
        <button
          onClick={sendMessage}
          disabled={isLoading}
          className="bg-blue-600 hover:bg-blue-700 text-white px-10 py-4 rounded-2xl font-medium transition"
        >
          Send
        </button>
      </div>
    </div>
  );
}
```

```tsx
'use client';
import { useState } from 'react';
import { supabase } from '@/lib/supabaseClient';
import { useRouter } from 'next/navigation';

export default function AuthPage() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [isLogin, setIsLogin] = useState(true);
  const router = useRouter();

  const handleAuth = async () => {
    const { error } = isLogin 
      ? await supabase.auth.signInWithPassword({ email, password })
      : await supabase.auth.signUp({ email, password });

    if (error) alert(error.message);
    else router.push('/');
  };

  return (
    <div className="max-w-md mx-auto mt-20 p-8 border rounded-2xl">
      <h2 className="text-3xl font-bold mb-6">{isLogin ? 'Login' : 'Sign Up'}</h2>
      <input type="email" placeholder="Email" value={email} onChange={(e) => setEmail(e.target.value)} className="w-full p-3 border rounded-xl mb-4" />
      <input type="password" placeholder="Password" value={password} onChange={(e) => setPassword(e.target.value)} className="w-full p-3 border rounded-xl mb-6" />
      <button onClick={handleAuth} className="w-full bg-blue-600 text-white py-3 rounded-xl">
        {isLogin ? 'Login' : 'Create Account'}
      </button>
      <button onClick={() => setIsLogin(!isLogin)} className="text-sm text-blue-600 mt-4">
        {isLogin ? "Don't have an account? Sign up" : "Already have an account? Login"}
      </button>
    </div>
  );
}
```

## Step 9: Document List with Summary Buttons

Create `components/DocumentList.tsx`:

```tsx
'use client';
import { useState, useEffect } from 'react';
import { supabase } from '@/lib/supabaseClient';
import { Progress } from '@/components/ui/progress';

export default function DocumentList() {
  const [documents, setDocuments] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchDocuments();
  }, []);

  const fetchDocuments = async () => {
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) return;

    const { data } = await supabase
      .from('documents')
      .select('id, name, status, summary, created_at')
      .eq('owner_id', user.id)
      .order('created_at', { ascending: false });

    setDocuments(data || []);
    setLoading(false);
  };

  const handleSummarize = async (docId: string) => {
    const { data: { user } } = await supabase.auth.getUser();
    const res = await fetch('/api/summarize', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ documentId: docId, userId: user?.id })
    });
    const { summary } = await res.json();
    alert(`Summary:\n\n${summary}`);
    fetchDocuments();
  };

  if (loading) return <p>Loading your documents...</p>;

  return (
    <div className="max-w-4xl mx-auto p-6">
      <h2 className="text-3xl font-bold mb-8">Your Research Documents</h2>
      <div className="grid gap-6">
        {documents.map((doc) => (
          <div key={doc.id} className="border rounded-2xl p-6 bg-white shadow-sm">
            <div className="flex justify-between items-start">
              <div>
                <h3 className="font-semibold text-lg">{doc.name}</h3>
                <p className="text-sm text-gray-500">Uploaded: {new Date(doc.created_at).toLocaleDateString()}</p>
              </div>
              <span className={`px-3 py-1 rounded-full text-xs font-medium capitalize ${doc.status === 'completed' ? 'bg-green-100 text-green-700' : 'bg-yellow-100 text-yellow-700'}`}>
                {doc.status}
              </span>
            </div>

            {doc.summary && (
              <div className="mt-4 p-4 bg-gray-50 rounded-xl text-sm">
                <strong>Summary:</strong> {doc.summary.substring(0, 280)}...
              </div>
            )}

            <div className="mt-6 flex gap-3">
              {doc.status === 'completed' && (
                <button onClick={() => handleSummarize(doc.id)} className="px-5 py-2 bg-blue-600 text-white rounded-xl hover:bg-blue-700">
                  Generate / View Summary
                </button>
              )}
              <button className="px-5 py-2 border border-gray-300 rounded-xl">Open in Chat</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
```
